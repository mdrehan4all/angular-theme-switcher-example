This is theme switcher example

Use following command to generate service

> ng generate service theme

Then add use following code

{{ "
// src/app/theme.service.ts
import { Injectable, Renderer2, RendererFactory2 } from '@angular/core';

@Injectable({
  providedIn: 'root'
})
export class ThemeService {
  private renderer: Renderer2;
  private currentTheme: 'day-theme' | 'night-theme' = 'day-theme';

  constructor(rendererFactory: RendererFactory2) {
    this.renderer = rendererFactory.createRenderer(null, null);
  }

  setTheme(theme: 'day-theme' | 'night-theme') {
    this.renderer.removeClass(document.body, this.currentTheme);
    this.currentTheme = theme;
    this.renderer.addClass(document.body, theme);
  }

  getTheme(): 'day-theme' | 'night-theme' {
    return this.currentTheme;
  }
}
" }}

Now use in any component as shown below:

{{"
import { Component } from '@angular/core';
import { ThemeService } from '../theme.service';

@Component({
  selector: 'app-home',
  templateUrl: './home.component.html',
  styleUrl: './home.component.scss'
})
export class HomeComponent {
  constructor(private themeService: ThemeService) {}

  toggleDayTheme() {
    this.themeService.setTheme('day-theme');
  }

  toggleNightTheme() {
    this.themeService.setTheme('night-theme');
  }
}
"}}

HTML:

<button (click)="toggleDayTheme()">Day Theme</button>
<button (click)="toggleNightTheme()"<Night Theme</button>
